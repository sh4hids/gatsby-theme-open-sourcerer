export const pageQuery = graphql`
  query {
    allYamlPage(filter: { pageType: { eq: "About" } }) {
      nodes {
        metaDescription
        pageType
        title
      }
    }
  }
`;

export default function About({ children, data }) {
  const pageData = data.allYamlPage.nodes[0] || {};
  return (
    <DefaultLayout
      title={pageData.title}
      description={pageData.metaDescription}
      url="/about/"
    >
      {children}
    </DefaultLayout>
  );
}
